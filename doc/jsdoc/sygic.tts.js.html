/**
 * JavaScript interface to abstract
 * the usage of the cordova Sygic Navigation plugin tts functionality.
 * @module Sygic.Tts
 */


var exec = require('cordova/exec');

/**
 * API for TTS.
 * 
 * @constructor
 */
function Tts (feature) {
    this.feature = feature;
}
/**
 * Uses the TTS engine, if installed, to speak the given text message. 
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {string}   args.text       - Text message to be read by TTS.
 * @param {Number}   args.background - Display option for action.
 */
Tts.prototype.playSound = function (success, error, args) {
    exec(success, error, this.feature, "playSound", [args]);
};

module.exports = Tts;

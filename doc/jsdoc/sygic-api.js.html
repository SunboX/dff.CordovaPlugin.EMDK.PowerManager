'use strict';

/**
 * Basic Sygic Navigation plugin functionality.
 *
 * @module com/dff/cordova/plugins/sygic/api
 */
 
var exec = require('cordova/exec');
var Dialog =     require('com.dff.cordova.plugins.sygic.api.dialog');
var Info =       require('com.dff.cordova.plugins.sygic.api.info');
var Itinerary =  require('com.dff.cordova.plugins.sygic.api.itinerary');
var Location =   require('com.dff.cordova.plugins.sygic.api.location');
var Maps =       require('com.dff.cordova.plugins.sygic.api.maps');
var Navigation = require('com.dff.cordova.plugins.sygic.api.navigation');
var Tts =        require('com.dff.cordova.plugins.sygic.api.tts');

/**
 * Basic Api functionality.
 *
 * @constructor
 * @param {string} feature - the Sygic Navigation feature to interact with.
 * Must be a property of {@link Sygic.ApplicationId}.
 */
function Api(feature) {
    this.feature = feature;
    this.dialog = new Dialog(this.feature);
    this.info = new Info(this.feature);
    this.itinerary = new Itinerary(this.feature);
    this.location = new Location(this.feature);
    this.maps = new Maps(this.feature);
    this.navigation = new Navigation(this.feature);
    this.tts = new Tts(this.feature);
}

/**
 * Show navigation application.
 *
 * @param {Function} success         - Callback if action is successful.
 * @param {Function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {Number}   args.background - Display option for action.
 */
Api.prototype.showApplication = function (success, error, args) {
    exec(success, error, this.feature, "showApplication", [args]);
};

/**
 * Ends navigation application. 
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {Number}   args.background - Display option for action.
 */
Api.prototype.endApplication = function (success, error, args) {
    exec(success, error, this.feature, "endApplication", [args]);
};

/**
 * Gets the application version and build number.
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {Number}   args.background - Display option for action.
 */
Api.prototype.appVersion = function (success, error, args) {
    exec(success, error, this.feature, "appVersion", [args]);
};

module.exports = Api;

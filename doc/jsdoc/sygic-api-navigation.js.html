'use strict';
/**
 * Sygic Navigation plugin tts functionality.
 *
 * @module com/dff/cordova/plugins/sygic/api/navigation
 */

var exec = require('cordova/exec');

/**
 * Api for Navigation
 *
 * @constructor
 * @param {string} feature - the Sygic Navigation feature to interact with.
 * Must be a property of {@link Sygic.ApplicationId}.
 */
function Navigation(feature) {
    this.feature = feature;
}

/**
 * Returns informations about actual GPS position.
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {boolean}  args.satInfo    - Whether function should return satellites information.
 * @param {Number}   args.background - Display option for action.
 */
Navigation.prototype.getActualGpsPosition = function (success, error, args) {
    exec(success, error, this.feature, "getActualGpsPosition", [args]);
};

/**
 * Starts navigation to given location.
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicAddress|SygicAddress}
 *
 * @param {function}                 success         - Callback if action is successful.
 * @param {function}                 error           - Callback if action is not successful.
 * @param {Object}                   args            - An object containing the parameters.
 * @param {SygicAddress}             args.address    - {@link module:com/dff/cordova/plugins/sygic/model~SygicAddress|SygicAddress} object containing iso, city, street, houseNo.
 * @param {boolean}                  args.postal     - Whether city represents postal or name.
 * @param {Number}                   args.flags      - Route computing properties.
 * @param {Number}                   args.background - Display option for action.
 */
Navigation.prototype.navigateToAddress = function (success, error, args) {
    exec(success, error, this.feature, "navigateToAddress", [args]);
};

/**
 * Returns informations about computed route. 
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {boolean}  args.extended   - Whether to return time and distance to each of the waypoints on the route
 * @param {Number}   args.background - Display option for action.
 */
Navigation.prototype.routeInfo = function (success, error, args) {
    exec(success, error, this.feature, "routeInfo", [args]);
};

/**
 * Starts navigation to given location.
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicWayPoint|SygicWayPoint}
 *
 * @param {function}                  success            - Callback if action is successful.
 * @param {function}                  error              - Callback if action is not successful.
 * @param {Object}                    args               - An object containing the parameters.
 * @param {SygicWayPoint}             args.wayPoint      - {@link module:com/dff/cordova/plugins/sygic/model~SygicWayPoint|SygicWayPoint}
 *                                                         which contains information baout the navigation target (strAddress, location).
 * @param {Number}                    args.flags         - Route computing properties.
 * @param {boolean}                   args.searchAddress - Whether address for location should be searched.
 * @param {Number}                    args.background    - Display option for action.
 */
Navigation.prototype.startNavigation = function (success, error, args) {
    exec(success, error, this.feature, "startNavigation", [args]);
};

/**
 * Stops navigation.
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {Number}   args.background - Display option for action.
 */
Navigation.prototype.stopNavigation = function (success, error, args) {
    exec(success, error, this.feature, "stopNavigation", [args]);
};

module.exports = Navigation;

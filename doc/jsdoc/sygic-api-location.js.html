'use strict';

/**
 * Sygic Navigation plugin location functionality.
 * @module com/dff/cordova/plugins/sygic/api/location
 */

var exec = require('cordova/exec');

/**
 * Api for Location
 *
 * @constructor
 * @param {string} feature - the Sygic Navigation feature to interact with.
 * Must be a property of {@link Sygic.ApplicationId}.
 */
function Location(feature) {
    this.feature = feature;
}
/**
 * Returns address corresponding to given location.
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition}
 *
 * @param {function} success            - Callback if action is successful.
 * @param {function} error              - Callback if action is not successful.
 * @param {Object} args                 - An object containing the parameters.
 * @param {SygicPosition} args.location - {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition} instance which contains
                                          location for which the address is searched. 
 * @param {Number} args.background      - Display option for action.
 */
Location.prototype.locationAddressInfo = function (success, error, args) {
    exec(success, error, this.feature, "locationAddressInfo", [args]);
};

/**
 * Returns location corresponding to given address. 
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicAddress|SygicAddress}
 *
 * @param {function}                 success          - Callback if action is successful.
 * @param {function}                 error            - Callback if action is not successful.
 * @param {Object}                   args             - An object containing the parameters.
 * @param {SygicAddress}             args.address     - {@link module:com/dff/cordova/plugins/sygic/model~SygicAddress|SygicAddress}
 *                                                      object containing iso, city, street, houseNo.
 * @param {boolean}                  args.postal      - Whether city represents postal or name.
 * @param {boolean}                  args.fuzzySearch - Whether the result will be the exact match or the best value match.
 * @param {Number}                   args.background  - Display option for action.
 */
Location.prototype.locationFromAddress = function (success, error, args) {
    exec(success, error, this.feature, "locationFromAddress", [args]);
};

/**
 * Returns list of locations corresponding to given address. 
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicAddress|SygicAddress}
 *
 * @param {function}                 success          - Callback if action is successful.
 * @param {function}                 error            - Callback if action is not successful.
 * @param {Object}                   args             - An object containing the parameters.
 * @param {SygicAddress}             args.address     - {@link module:com/dff/cordova/plugins/sygic/model~SygicAddress|SygicAddress}
 *                                                      object containing iso, city, street, houseNo.
 * @param {boolean}                  args.postal      - Whether city represents postal or name.
 * @param {boolean}                  args.fuzzySearch - Whether the result will be the exact match or the best value match.
 * @param {Number}                   args.background  - Display option for action.
 */
Location.prototype.locationFromAddressEx = function (success, error, args) {
    exec(success, error, this.feature, "locationFromAddressEx", [args]);
};

/**
 * Returns road info corresponding to given location.
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition}
 *
 * @param {function} success            - Callback if action is successful.
 * @param {function} error              - Callback if action is not successful.
 * @param {Object} args                 - An object containing the parameters.
 * @param {SygicPosition} args.location - {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition}
                                          which contains location for which the road info is searched. 
 * @param {Number} args.background      - Display option for action.
 */
Location.prototype.locationRoadInfo = function (success, error, args) {
    exec(success, error, this.feature, "locationRoadInfo", [args]);
};

module.exports = Location;

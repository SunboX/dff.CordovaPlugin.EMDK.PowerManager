'use strict';

/**
 * Sygic Navigation plugin info functionality.
 *
* @module com/dff/cordova/plugins/sygic/api/dialog
 */

var exec = require('cordova/exec');
/**
 * Api for Dialog
 *
 * @constructor
 * @param {string} feature - the Sygic Navigation feature to interact with.
 * Must be a property of {@link Sygic.ApplicationId}.
 */
function Dialog(feature) {
    this.feature = feature;
}
/**
 * Shows a flash message.
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {string}   args.message    - Text of the message to be shown. 
 * @param {Number}   args.background - Display option for action.
 */
Dialog.prototype.flashMessage = function (success, error, args) {
    exec(success, error, this.feature, "flashMessage", [args]);
};

/**
 * Shows dialog window with a message.
 *
 * @param {function} success              - Callback if action is successful.
 * @param {function} error                - Callback if action is not successful.
 * @param {Object}   args                 - An object containing the parameters.
 * @param {string}   args.message         - Text of the message to be shown. 
 * @param {Number}   args.buttons         - Button to be show to the user ({@link model.SygicStopOffPoint.pointType}).
 * @param {boolean}  args.waitForFeedback - Whether Application returns reply before user click the button. 
 * @param {Number}   args.background      - Display option for action.
 */
Dialog.prototype.showMessage = function (success, error, args) {
    exec(success, error, this.feature, "showMessage", [args]);
};

module.exports = Dialog;

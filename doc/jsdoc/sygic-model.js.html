'use strict';

/**
 * Sygic Navigation plugin model.
 *
 * @module com/dff/cordova/plugins/sygic/model
 */

var exec = require('cordova/exec');

/**
 * A SygicAddress holds the required parts of a address used by Sygic Navigation API.
 *
 * @constructor
 * @param {string} iso         - iso code of the country
 * @param {string|Number} city - name or postal code of the city
 * @param {string} street      - name of the street
 * @param {string} houseno     - house number
 */
function SygicAddress(iso, city, street, houseno) {
    this.iso = iso;
    this.city = city;
    this.street = street;
    this.houseno = houseno;
}

/**
 * Represents a location (Position) on map.
 *
 * @constructor    
 * @param {Number} lon - geographical longitude in world degrees.
 * @param {Number} lat - geographical latitude in world degrees.
 */
function SygicPosition(lon, lat) {
    this.lon = lon;
    this.lat = lat;
}

/**
 * Bounds a rectangle on map.
 *
 * @constructor
 * @param {Number} top    - top border of rectangle as geographical longitude in world degrees.
 * @param {Number} right  - right border of rectangle as geographical latitude in world degrees.
 * @param {Number} bottom - bottom border of rectangle as geographical longitude in world degrees.
 * @param {Number} left   - left border of rectangle as geographical latitude in world degrees.
 */
function SygicRectangle(top, right, bottom, left) {
    this.top = top;
    this.right = right;
    this.bottom = bottom;
    this.left = left;
}

/**
 * Specifies the location of one point on map.
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition}
 *
 * @constructor
 * @param {string}        strAddress - name describing the location.
 * @param {SygicPosition} location   - location ({@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition})
 *                                      of one point on map.
 */
function SygicWayPoint(strAddress, location) {
    this.strAddress = strAddress;
    this.location = location;
}

/**
 * Represent a stop of an itinerary.
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition}
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicStopOffPoint.pointType|pointType}
 *
 * @constructor
 * @param {boolean} isSearchAddress - Flag determining whether address should be searched (for example when adding waypoint). 
 * @param {boolean} isVisited       - Flag determinig whether waypoint was already visited.
 * @param {Number}  pointType       - {@link module:com/dff/cordova/plugins/sygic/model~SygicStopOffPoint.pointType|pointType} of stop.
 * @param {SygicPosition} location  - Location ({@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition})
 *                                    of stop.
 * @param {Number} offset           - Road offset from our map, where the waypoint is located.
 *                                    Use conversion table to obtain the offset from map provider segment ID.
 *                                    When using offset, the ISO code has to be used. 
 * @param {Number} id               - Unique id of stop.
 * @param {string} iso              - Iso code of country the stop is located.
 * @param {string} caption          - Name of stop.
 * @param {string} address          - Address of stop.
 */
function SygicStopOffPoint(isSearchAddress, isVisited, pointType, location, offset, id, iso, caption, address) {
    this.isSearchAddress = isSearchAddress;
    this.isVisited = isVisited;
    this.pointType = pointType;
    this.location = location;
    this.offset = offset;
    this.id = id;
    this.iso = iso;
    this.caption = caption;
    this.address = address;
}

/**
 * Types of stops in itinerary.
 *
 * @property {Number} VIA - Viapoint.
 * @property {Number} FINISH - Finish.
 * @property {Number} START - Start.
 * @property {Number} INVISIBLE - Invisible. 
 */
SygicStopOffPoint.pointType = {
    VIA: 1,
    FINISH: 2,
    START: 3,
    INVISIBLE: 4
};

module.exports = {
    SygicAddress: SygicAddress,
    SygicPosition: SygicPosition,
    SygicRectangle: SygicRectangle,
    SygicWayPoint: SygicWayPoint,
    SygicStopOffPoint: SygicStopOffPoint
};

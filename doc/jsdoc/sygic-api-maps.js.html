'use strict';

/**
 * Sygic Navigation plugin maps functionality.
 *
* @module com/dff/cordova/plugins/sygic/api/maps
 */

var exec = require('cordova/exec');

/**
 * Api for Maps
 *
 * @constructor
 * @param {string} feature - the Sygic Navigation feature to interact with.
 * Must be a property of {@link Sygic.ApplicationId}.
 */
function Maps(feature) {
    this.feature = feature;
}

/**
 * Shows taken location on map.
 * 
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition}
 *
 * @param {function}                  success         - Callback if action is successful.
 * @param {function}                  error           - Callback if action is not successful.
 * @param {Object}                    args            - An object containing the parameters.
 * @param {SygicPosition}             args.location   - Instance of  {@link module:com/dff/cordova/plugins/sygic/model~SygicPosition|SygicPosition}
 *                                                      which contains location which will be shown on BrowseMap. 
 * @param {Number}                    args.zoom       - Distance (in meters) form ground.
 *                                                      Range of this parameter should be between 50 and 30000000. 
 * @param {Number}                    args.background - Display option for action.
 */
Maps.prototype.showCoordinatesOnMap = function (success, error, args) {
    exec(success, error, this.feature, "showCoordinatesOnMap", [args]);
};

/**
 * Shows given rectangle on the map.
 * Zoom will be adjusted so that entire rectangle will be shown. 
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicRectangle|SygicRectangle}
 *
 * @param {function}                   success         - Callback if action is successful.
 * @param {function}                   error           - Callback if action is not successful.
 * @param {Object}                     args            - An object containing the parameters.
 * @param {SygicRectangle}             args.rectangle  - {@link module:com/dff/cordova/plugins/sygic/model~SygicRectangle|SygicRectangle}
 *                                                        to show on the map.
 * @param {Number}                     args.background - Display option for action.
 */
Maps.prototype.showRectangleOnMap = function (success, error, args) {
    exec(success, error, this.feature, "showRectangleOnMap", [args]);
};

module.exports = Maps;

/**
 * JavaScript interface to abstract
 * the usage of the cordova Sygic Navigation plugin.
 *
 * @module com/dff/cordova/plugins/sygic
 */

'use strict';

var exec = require('cordova/exec');
var cordova = require('cordova');
var channel = require('cordova/channel');

var Model = require('com.dff.cordova.plugins.sygic.model');
var Api = require('com.dff.cordova.plugins.sygic.api');

var sygicInstance = {};

/**
 * Exception thrown by the Sygic JavaScript interface.
 * @constructor
 * @param {string} message - the message describing the exception
 */
function SygicException(message) {
    this.message = message;
    this.name = "SygicException";
}

/**
 * Abstract super class for the Sygic Navigation features
 * Drive, Fleet, Taxi and Truck.
 * Provides functions to interact with a Sygic Navigation app.
 *
 * @constructor
 * @param {string} feature - the Sygic Navigation feature to interact with.
 * Must be a property of {@link Sygic.ApplicationId}.
 */
function Sygic(feature) {
    var self = this;
    this.feature = feature;
    this.api = new Api(this.feature);

    channel.onCordovaReady.subscribe(function () {
        self.onServiceEvent(function (event) {
            cordova.fireDocumentEvent(self.feature + "ServiceEvent", event);
        }, function (error) {
            console.log(JSON.stringify(error));
        });

        self.onAppEvent(function (event) {
            cordova.fireDocumentEvent(self.feature + "AppEvent", event);
        }, function (error) {
            console.log(JSON.stringify(error));
        });
    });
}

/**
 * Returns the Sygic Navigation feature name the instance interacts with.
 *
 * @returns {string} Sygic Navigation feature name.
 */
Sygic.prototype.getFeature = function () {
    return this.feature;
};

// Api Events

Sygic.prototype.onServiceEvent = function (success, error) {
    exec(success, error, this.feature, "onServiceEvent", []);
};

Sygic.prototype.onAppEvent = function (success, error) {
    exec(success, error, this.feature, "onAppEvent", []);
};

/**
 * The start of working with the Sygic Navigation plugin.
 * Returns a instance to interact with the specified Sygic Navigation feature.
 *
 * @param {string} applicationId - the {@link Sygic.ApplicationId} of the feature to interact with.
 * @returns {Sygic} an instance of the Sygic Plugin interface.
 */
Sygic.getInstance = function (applicationId) {
    if (!(sygicInstance[applicationId] instanceof Sygic)) {
        switch (applicationId) {
        case Sygic.ApplicationId.SygicDrive:
            sygicInstance[applicationId] = new SygicDrive();
            break;
        case Sygic.ApplicationId.SygicFleet:
            sygicInstance[applicationId] = new SygicFleet();
            break;
        case Sygic.ApplicationId.SygicTaxi:
            sygicInstance[applicationId] = new SygicTaxi();
            break;
        case Sygic.ApplicationId.SygicTruck:
            sygicInstance[applicationId] = new SygicTruck();
            break;
        default:
            throw new SygicException("Unknown ApplicationId: " + applicationId);
        }
    }

    return sygicInstance[applicationId];
};

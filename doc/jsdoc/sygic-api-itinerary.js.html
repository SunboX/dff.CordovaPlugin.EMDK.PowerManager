'use strict';

/**
 * Sygic Navigation plugin itinerary functionality.
* @module com/dff/cordova/plugins/sygic/api/itinerary
 */

var exec = require('cordova/exec');

/**
 * Api for Itinerary
 *
 * @constructor
 * @param {string} feature - the Sygic Navigation feature to interact with.
 * Must be a property of {@link Sygic.ApplicationId}.
 */
function Itinerary(feature) {
    this.feature = feature;
}
/**
 * Adds new waypoint to taken itinerary. 
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicStopOffPoint|SygicStopOffPoint}
 *
 * @param {function} success              - Callback if action is successful.
 * @param {function} error                - Callback if action is not successful.
 * @param {Object}   args                 - An object containing the parameters.
 * @param {string}   args.itineraryName   - Name of itinerary where new item will be added.
 * @param {StopOffPoint} args.stopOffPoint - {@link module:com/dff/cordova/plugins/sygic/model~SygicStopOffPoint|SygicStopOffPoint}
 *                                          containing information about the waypoint to be added.
 * @param {Number}   args.index           - Position where waypoint will be inserted.
 * @param {Number}   args.background      - Display option for action.
 */
Itinerary.prototype.addEntryToItinerary = function (success, error, args) {
    exec(success, error, this.feature, "addEntryToItinerary", [args]);
};

/**
 * Creates itinerary and fills it with given waypoints. 
 *
 * @see {@link module:com/dff/cordova/plugins/sygic/model~SygicStopOffPoint|SygicStopOffPoint}
 *
 * @param {function}       success            - Callback if action is successful.
 * @param {function}       error              - Callback if action is not successful.
 * @param {Object}         args               - An object containing the parameters.
 * @param {string}         args.itineraryName - Name of itinerary that will be created.
 * @param {StopOffPoint[]} args.stopOffPoints - List of {@link module:com/dff/cordova/plugins/sygic/model~SygicStopOffPoint|SygicStopOffPoint}
 *                                              of itinerary.
 * @param {Number}         args.background    - Display option for action.
 */
Itinerary.prototype.addItinerary = function (success, error, args) {
    exec(success, error, this.feature, "addItinerary", [args]);
};

/**
 * Deletes waypoint from itinerary at given position. 
 *
 * @param {function} success            - Callback if action is successful.
 * @param {function} error              - Callback if action is not successful.
 * @param {Object}   args               - An object containing the parameters.
 * @param {string}   args.itineraryName - Name of itinerary where item will be removed.
 * @param {Number}   args.index         - Position of waypoint that will be removed.
 * @param {Number}   args.background    - Display option for action.
 */
Itinerary.prototype.deleteEntryInItinerary = function (success, error, args) {
    exec(success, error, this.feature, "deleteEntryInItinerary", [args]);
};

/**
 * Deletes given itinerary. 
 *
 * @param {function} success            - Callback if action is successful.
 * @param {function} error              - Callback if action is not successful.
 * @param {Object}   args               - An object containing the parameters.
 * @param {string}   args.itineraryName - Name of itinerary to be removed.
 * @param {Number}   args.background    - Display option for action.
 */
Itinerary.prototype.deleteItinerary = function (success, error, args) {
    exec(success, error, this.feature, "deleteItinerary", [args]);
};

/**
 * Returns list of waypoints in itinerary. 
 *
 * @param {function} success            - Callback if action is successful.
 * @param {function} error              - Callback if action is not successful.
 * @param {Object}   args               - An object containing the parameters.
 * @param {string}   args.itineraryName - Name of itinerary from which waypoints will be retrieved.
                                          If the name of itinerary is "default",
                                          function returns the list of waypoints from current route. 
 * @param {Number}   args.background    - Display option for action.
 */
Itinerary.prototype.getItineraryList = function (success, error, args) {
    exec(success, error, this.feature, "getItineraryList", [args]);
};

/**
 * Computes route based on taken itinerary.
 * If Navigation has valid GPS signal it starts navigation.
 *
 * @param {function} success            - Callback if action is successful.
 * @param {function} error              - Callback if action is not successful.
 * @param {Object}   args               - An object containing the parameters.
 * @param {string}   args.itineraryName - Name of the itinerary which will be used to compute route.
 * @param {Number}   args.flags         - Parameter flags determines route computing properties.
 * @param {Number}   args.background    - Display option for action.
 */
Itinerary.prototype.setRoute = function (success, error, args) {
    exec(success, error, this.feature, "setRoute", [args]);
};

/**
 * Skips the next waypoint. 
 *
 * @param {function} success         - Callback if action is successful.
 * @param {function} error           - Callback if action is not successful.
 * @param {Object}   args            - An object containing the parameters.
 * @param {Number}   args.background - Display option for action.
 */
Itinerary.prototype.skipNextWayPoint = function (success, error, args) {
    exec(success, error, this.feature, "skipNextWayPoint", [args]);
};

module.exports = Itinerary;
